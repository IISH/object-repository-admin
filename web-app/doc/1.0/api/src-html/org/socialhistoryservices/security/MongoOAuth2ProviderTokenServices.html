041    /**
042     * OAuth2 provider of tokens. Made for MongoDB

049        private static final String OAUTH_ACCESS_TOKEN = "oauth_access_token";

051        private final ConcurrentHashMap<String, OAuth2AccessToken> accessTokenStore = new ConcurrentHashMap<String, OAuth2AccessToken>();
052        private final ConcurrentHashMap<String, OAuth2Authentication> authenticationTokenStore = new ConcurrentHashMap<String, OAuth2Authentication>();
053        private final ConcurrentHashMap<String, Long> expirationTokenStore = new ConcurrentHashMap<String, Long>();
054        private long sliderExpiration = 30000; // Thirty seconds

066        public OAuth2AccessToken selectKeys(String username) {
067    
068            // select token, authentication from oauth_access_token
069            final BasicDBObject query = new BasicDBObject("name", username);
070            final DBCollection collection = getCollection(OAUTH_ACCESS_TOKEN);
071            DBObject document = collection.findOne(query);
072            OAuth2AccessToken token = (document == null)
073                    ? null
074                    : (OAuth2AccessToken) deserialize((byte[]) document.get("token"));
075            return token;
076        }

078        public OAuth2AccessToken recreateRefreshAccessToken(String refreshTokenValue) throws AuthenticationException {
079    
080            if (!isSupportRefreshToken()) {
081                throw new InvalidTokenException("Invalid refresh token: " + refreshTokenValue);
082            }
083    
084            ExpiringOAuth2RefreshToken refreshToken = readRefreshToken(refreshTokenValue);
085            if (refreshToken == null) {
086                throw new InvalidTokenException("Invalid refresh token: " + refreshTokenValue);
087            } else if (isExpired(refreshToken)) {
088                removeRefreshToken(refreshTokenValue);
089                throw new InvalidTokenException("Invalid refresh token: " + refreshToken);
090            }
091            removeAccessTokenUsingRefreshToken(refreshTokenValue); //clear out any access tokens already associated with the refresh token.
092    
093            final SecurityContext context = SecurityContextHolder.getContext();
094            Authentication userAuthentication = context.getAuthentication();
095            OAuth2Authentication authentication = clientAuthentication(userAuthentication);
096    
097            if (!isReuseRefreshToken()) {
098                removeRefreshToken(refreshTokenValue);
099                refreshToken = createRefreshToken(authentication);
100            }
101    
102            return createAccessToken(authentication, refreshToken);
103        }

105        @Override
106        protected void storeAccessToken(OAuth2AccessToken token, OAuth2Authentication authentication) {
107    
108            // "insert into oauth_access_token (token_id, token, authentication, refresh_token) values (?, ?, ?, ?)";
109            String refreshToken = null;
110            if (token.getRefreshToken() != null) {
111                refreshToken = token.getRefreshToken().getValue();
112            }
113            final BasicDBObject document = new BasicDBObject();
114            document.put("token_id", token.getValue());
115            document.put("token", serialize(token));
116            document.put("authentication", serialize(authentication));
117            document.put("refresh_token", refreshToken);
118            document.put("name", authentication.getName());
119            final DBCollection collection = getCollection(OAUTH_ACCESS_TOKEN);
120            collection.insert(document);
121    
122    
123        }
124    
125        @Override
126        protected OAuth2AccessToken readAccessToken(String tokenValue) {
127    
128            OAuth2AccessToken accessToken = (isFresh(tokenValue))
129                    ? this.accessTokenStore.get(tokenValue)
130                    : null;
131            if (accessToken == null) {
132                // select token_id, token from oauth_access_token where token_id = ?
133                final BasicDBObject query = new BasicDBObject();
134                query.put("token_id", tokenValue);
135                final DBCollection collection = getCollection(OAUTH_ACCESS_TOKEN);
136                DBObject document = collection.findOne(query);
137                if (document == null) {
138                } else {
139                    accessToken = deserialize((byte[]) document.get("token"));
140                    this.accessTokenStore.put(tokenValue, accessToken);
141                    expiration(tokenValue);
142                }
143            }
144            return accessToken;
145        }
146    
147        @Override
148        protected void removeAccessToken(String tokenValue) {
149    
150            final BasicDBObject query = new BasicDBObject();
151            query.put("token_id", tokenValue);
152            final DBCollection collection = getCollection(OAUTH_ACCESS_TOKEN);
153            collection.remove(query);
154            this.accessTokenStore.remove(tokenValue);
155        }
156    
157        @Override
158        protected OAuth2Authentication readAuthentication(OAuth2AccessToken token) {
159    
160            final String tokenValue = token.getValue();
161            OAuth2Authentication authentication = (isFresh(tokenValue))
162                    ? this.authenticationTokenStore.get(tokenValue)
163                    : null;
164            if (authentication == null) {
165                // select token_id, authentication from oauth_access_token where token_id = ?
166                final BasicDBObject query = new BasicDBObject();
167                query.put("token_id", token.getValue());
168                final DBCollection collection = getCollection(OAUTH_ACCESS_TOKEN);
169                final DBObject document = collection.findOne(query);
170                if (document == null) {
171                } else {
172                    authentication = deserialize((byte[]) document.get("authentication"));
173                    this.authenticationTokenStore.put(tokenValue, authentication);
174                    expiration(tokenValue);
175                }
176            }
177            return authentication;
178        }
179    
180        @Override
181        protected void storeRefreshToken(ExpiringOAuth2RefreshToken refreshToken, OAuth2Authentication authentication) {
182    
183            // insert into oauth_refresh_token (token_id, token, authentication) values (?, ?, ?)
184            final BasicDBObject document = new BasicDBObject();
185            document.put("token_id", refreshToken.getValue());
186            document.put("token", serialize(refreshToken));
187            document.put("authentication", serialize(authentication));
188            final DBCollection collection = getCollection(OAUTH_REFRESH_TOKEN);
189            collection.insert(document);
190        }
191    
192        @Override
193        protected ExpiringOAuth2RefreshToken readRefreshToken(String token) {
194    
195            // select token_id, token from oauth_refresh_token where token_id = ?
196            ExpiringOAuth2RefreshToken refreshToken = null;
197            final BasicDBObject query = new BasicDBObject();
198            query.put("token_id", token);
199            final DBCollection collection = getCollection(OAUTH_REFRESH_TOKEN);
200            final DBObject document = collection.findOne(query);
201            if (document == null) {
202            } else {
203                refreshToken = deserialize((byte[]) document.get("token"));
204            }
205            return refreshToken;
206        }

244        private void expiration(String tokenValue) {
245            this.expirationTokenStore.put(tokenValue, new Date().getTime() + sliderExpiration);
246        }
247    
248        /**
249         * Determines if we can use the cache...
250         *
251         * @param tokenValue
252         * @return
253         */
254        private boolean isFresh(String tokenValue) {
255    
256            long expiration = (expirationTokenStore.containsKey(tokenValue))
257                    ? expirationTokenStore.get(tokenValue)
258                    : 0;
259            long time = new Date().getTime();
260            if (expiration > time) {
261                this.expirationTokenStore.put(tokenValue, time + sliderExpiration);
262                return true;
263            }
264            expirationTokenStore.remove(tokenValue);
265            accessTokenStore.remove(tokenValue);
266            authenticationTokenStore.remove(tokenValue);
267            return false;
268        }

270        public OAuth2AccessToken createToken(Authentication userAuthentication) {
271    
272            OAuth2Authentication authentication = clientAuthentication(userAuthentication);
273            return createAccessToken(authentication);
274        }

276        private OAuth2Authentication clientAuthentication(Authentication userAuthentication) {
277            AuthorizedClientAuthenticationToken clientAuthentication = new AuthorizedClientAuthenticationToken(clientId, clientSecret, scope, authorities);
278            return new OAuth2Authentication(clientAuthentication, userAuthentication);
279        }

320        public static byte[] serialize(Object state) {
321            ObjectOutputStream oos = null;
322            try {
323                ByteArrayOutputStream bos = new ByteArrayOutputStream(512);
324                oos = new ObjectOutputStream(bos);
325                oos.writeObject(state);
326                oos.flush();
327                return bos.toByteArray();
328            } catch (IOException e) {
329                throw new IllegalArgumentException(e);
330            } finally {
331                if (oos != null) {
332                    try {
333                        oos.close();
334                    } catch (IOException e) {
335                        // eat it
336                    }
337                }
338            }
339        }
340    
341        public static <T> T deserialize(byte[] byteArray) {
342            ObjectInputStream oip = null;
343            try {
344                oip = new ObjectInputStream(new ByteArrayInputStream(byteArray));
345                return (T) oip.readObject();
346            } catch (IOException e) {
347                throw new IllegalArgumentException(e);
348            } catch (ClassNotFoundException e) {
349                throw new IllegalArgumentException(e);
350            } finally {
351                if (oip != null) {
352                    try {
353                        oip.close();
354                    } catch (IOException e) {
355                        // eat it
356                    }
357                }
358            }
359        }
