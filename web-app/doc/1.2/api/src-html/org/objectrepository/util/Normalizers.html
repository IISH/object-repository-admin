001    package org.objectrepository.util;
002    
003    import org.apache.commons.io.FilenameUtils;
004    
005    import java.io.File;
006    
007    final public class Normalizers {
008    
009        public static String normalize(File file) {
010            return normalize(file.getAbsolutePath());
011        }
012    
013        /**
014         * Ensures all fileformats are absolute and with forward slash and without dot or driveletter
015         *
016         * @param absolutePath
017         * @return
018         */
019        public static String normalize(String absolutePath) {
020    
021            String s = absolutePath.trim().replaceAll("\\n|\\r|\\t", "");
022            return FilenameUtils.separatorsToUnix(s.replaceFirst("[A-Za-z][:]", ""));
023        }
024    
025        /**
026         * toRelative
027         * <p/>
028         * Expresses the location of the file as a relative path to the fileset and starting from the first folder.
029         * <p/>
030         * The relative path of a file is then: first folder of fileSet + location
031         * We do not write an absolute location to preserve the file view of the staging area.
032         * <p/>
033         * That is: the sftp client's perspective of the staging area is identical to that of the file location.
034         *
035         * @param fileSet
036         * @param location
037         * @return
038         */
039        public static String toRelative(String fileSet, String location) {
040            String _fileSet = normalize(fileSet);
041            String _location = normalize(location);
042            File folder = new File(fileSet);
043            return "/" + folder.getName() + _location.substring(_fileSet.length());
044        }
045    
046        public static String toRelative(String fileSet, File file) {
047            return toRelative(fileSet, file.getPath());
048        }
049    
050        /**
051         * Converts a relative file specification into an absolute fs path.
052         *
053         * @return
054         */
055        public static File toAbsolute(String fileSet, String location) {
056            return new File(FilenameUtils.getFullPath(fileSet) + location);
057        }
058    
059        public static boolean isEmpty(String text) {
060            return (text == null) ? true : text.isEmpty();
061        }
062    
063    }
